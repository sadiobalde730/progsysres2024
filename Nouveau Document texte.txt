#include <stdio.h>
#include <stdlib.h>
void matrice_aleatoire(int **mat , int taille){
    for(int i=0 ; i<taille ; i++)
        for(int j=0 ; j<taille ; j++)
            mat[i][j] = rand() ;
}

int main(){
    int **mat ;
    matrice_aleatoire(mat,3);
    for(int i=0 ; i<3 ; i++)
        for(int j=0 ; j<3 ; j++)
            printf("%d\t",mat[i][j]);
        printf("\n");
    return 0 ;

}
